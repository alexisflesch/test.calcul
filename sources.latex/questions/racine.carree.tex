
\begin{qcm}{racine.simple}
    \begin{question}{racine.simple.1.a}
         Simplifier \(\sqrt{20}\).
         \vspace{-1.5ex}
         \begin{multicols}{3}
         \begin{reponses}
            \bonne{\(2\sqrt{5}\)}
            \mauvaise{\(5\sqrt{2}\)}
            \mauvaise{\(\sqrt{2}+\sqrt{5}\)}
            \mauvaise{\(5+\sqrt{2}\)}
            \mauvaise{\(2+\sqrt{5}\)}
         \end{reponses}
         \end{multicols}
    \end{question}
\end{qcm}

\begin{qcm}{racine.moyen}
    \begin{question}{racine.moyen.1.a}
      Soient \(x\) et \(y\) des réels tels que \(x^2-y^2\geq 0\). Alors, \(\sqrt{x^2-y^2}=\ldots\)
         \vspace{-1.5ex}
         \begin{multicols}{2}
         \begin{reponses}
            \mauvaise{\(x-y\)}
            \mauvaise{\(|x|-|y|\)}
            \mauvaise{\(|x|+|y|\)}
            \lastchoices
            \bonne{Aucune des réponses précédentes}
         \end{reponses}
         \end{multicols}
    \end{question}
\end{qcm}
