\begin{qcm}{autres.comparaisons}
	\begin{question}{autres.comparaisons.1.n}
		Soit \(0<a<1\). Parmi les 4 nombres réels suivants, cocher celui qui est le plus grand :
		\vspace{-0.2cm}
		\begin{multicols}{2}
			\begin{reponses}
				\bonne{\(\sqrt{a}\)}
				\mauvaise{\(a\)}
				\mauvaise{\(a^2\)}
				\mauvaise{\(a\sqrt{a}\)}
			\end{reponses}
		\end{multicols}
	\end{question}
\end{qcm}
