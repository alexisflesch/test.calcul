\begin{qcm}{signe.polynome}
    \begin{question}{signe.polynome.1.a}
         Quel est l'ensemble des solutions réelles de \(x^2+x-2\geqslant 0\)?
         \vspace{0.1cm}
         \begin{reponses}
            \bonne{\(]-\infty,-2]\cup[1,+\infty[\)}
            \mauvaise{\(]-\infty,-1]\cup[2,+\infty[\)}
            \mauvaise{\([0,1]\)}
            \mauvaise{\([-2,1]\)}
            \mauvaise{\([-1,2]\)}
         \end{reponses}
         \vspace{0.4cm}
    \end{question}
\end{qcm}
