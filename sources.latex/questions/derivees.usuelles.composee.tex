         Posons pour tout nombre réel \(x\), \( f(x)=\mathrm{e}^{-x} \). Alors \( f'(x)=\ldots\)

	    \begin{reponses}
            \lastchoices
	          \mauvaise{\(-\ln(x)\)}
              \mauvaise{\( -\mathrm{e}^{x} \)}
              \mauvaise{\( \mathrm{e}^{-x} \)}
              \bonne{\( - \mathrm{e}^{-x} \)}
              \mauvaise{\( \dfrac{1}{x} \)}
  		 
             \phantom{foo}
          \end{reponses}
         \end{multicols}
    \end{question}
